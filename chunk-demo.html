Usage Reference
<!-- ① 引用元件 --> <script src="chunk-demo.js"></script> <!-- ② 基本用法 --> <chunk-demo theme="sky" sentence="She {0} every morning." show-preview="true" chunks='[{"id":0,"icon":"☀️","label":"Routine","levels":[ {"level":1,"text":"goes for a walk"}, {"level":2,"text":"hits the jogging track"}, {"level":3,"text":"heads out for a brisk morning run"} ]}]' ></chunk-demo> <!-- ③ 外框粗細與樣式（HTML 屬性，最高優先權） --> <chunk-demo theme="indigo" border-width="2px" border-style="dashed" sentence="..." chunks='[...]' ></chunk-demo> <!-- ④ 全域設定（放在 chunk-demo.js 載入之前） --> <script> window.ChunkDemoConfig = { defaultTheme: 'sky', chunkBorderWidth: '1.5px', // 全域外框粗細 chunkBorderStyle: 'solid', // solid | dashed | dotted | double previewBorderColor: null, // null = 自動跟隨 theme 色 themes: { brand: { border:'#FF5733', text:'#FF5733', bg:'rgba(255,87,51,.09)' } }, levelDotColors: { 1:'#40C99A', 2:'#DECA4B', 3:'#C3A5E5' }, }; </script> // 優先權（低 → 高）： // DEFAULTS → ChunkDemoConfig → data-config attr → border-width/border-style attr
