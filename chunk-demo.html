子元素語法（推薦） 
<!-- sentence 中 {n} 對應 cd-chunk 的 id 或出現順序 -->
<chunk-demo
  theme="sky"
  sentence="She {0} every morning to {1}."
  show-preview="true">

  <cd-chunk id="0" icon="☀️" label="Activity">
    <cd-level>goes for a walk</cd-level>
    <cd-level>hits the jogging track</cd-level>
    <cd-level>heads out for a brisk morning run
               along the riverbank trail</cd-level>
  </cd-chunk>

  <cd-chunk id="1" icon="🎯">
    <!-- label 省略 → 下拉標頭不顯示 -->
    <cd-level>stay fit</cd-level>
    <cd-level>keep herself in shape</cd-level>
    <cd-level>maintain a healthy and active lifestyle</cd-level>
  </cd-chunk>

</chunk-demo>
Preview 顏色自訂（data-config 或全域 ChunkDemoConfig） 
<!-- per-element：透過 data-config 覆蓋 -->
<chunk-demo theme="orange" show-preview="true"
  data-config='{"previewTextColor":"#EDA109","previewBg":"rgba(237,161,9,.07)"}'
  sentence="..." chunks='[...]'></chunk-demo>

// 全域設定
window.ChunkDemoConfig = {
  previewBorderColor: null,              // null = 自動跟隨 theme 色
  previewTextColor:   null,              // null = 預設 #7a8078
  previewBg:          null,              // null = 預設 rgba(255,255,255,.035)
};
cd-level 內含 HTML 標籤 
<!-- cd-level 的文字內容以 innerHTML 解析，可使用任意 HTML -->
<cd-chunk id="0" icon="🌟">
  <cd-level><strong>bright</strong></cd-level>
  <cd-level>truly <em>radiant</em> and uplifting</cd-level>
  <cd-level>wonderfully <strong>vivid</strong> and
             <em>infectiously</em> positive outlook on life</cd-level>
</cd-chunk>

<!-- HTML 同時顯示於：按鈕文字、下拉選單、整句 preview -->
<!-- ⚠️ 勿將使用者輸入直接插入 cd-level，需自行做 sanitize -->
JSON Attribute 語法（向下相容） 
<chunk-demo theme="sky" sentence="She {0} every morning."
  chunks='[{"id":0,"icon":"☀️","label":"Activity","levels":[
    {"level":1,"text":"goes for a walk"},
    {"level":2,"text":"hits the jogging track"},
    {"level":3,"text":"heads out for a brisk morning run"}
  ]}]'></chunk-demo>

// 優先序：cd-chunk 子元素 > chunks JSON 屬性
